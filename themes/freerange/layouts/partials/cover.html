{{- /* Renders a post's cover illustration as a responsive WebP , by convention from the post filename: assets/images/covers/.{webp,png}. WebP masters are preferred; PNG is a fallback for un-migrated covers. Outputs nothing when no cover exists. Params (dict): page (required) the Page class css class (default "card-cover") sizes value (default full content width) priority true = LCP hero: fetchpriority=high, eager; false = lazy-loaded */ -}} {{- $page := .page -}} {{- $class := .class | default "card-cover" -}} {{- $sizes := .sizes | default "(min-width: 1100px) 1040px, 100vw" -}} {{- $priority := .priority | default false -}} {{- if $page.Params.cover -}} {{- else if $page.File -}} {{- $base := $page.File.ContentBaseName -}} {{- $src := or (resources.Get (printf "images/covers/%s.webp" $base)) (resources.Get (printf "images/covers/%s.png" $base)) -}} {{- with $src -}} {{- /* WebP renditions only at widths the source can supply (no upscaling). */ -}} {{- $chosen := slice -}} {{- range (slice 480 768 1024 1280 1600) -}} {{- if lt . $src.Width -}}{{- $chosen = $chosen | append . -}}{{- end -}} {{- end -}} {{- $chosen = $chosen | append $src.Width -}} {{- $srcset := slice -}} {{- range $chosen -}} {{- $r := $src.Resize (printf "%dx webp" .) -}} {{- $srcset = $srcset | append (printf "%s %dw" $r.RelPermalink .) -}} {{- end -}} {{- $fallback := $src.Resize (printf "%dx webp" (index $chosen (sub (len $chosen) 1))) -}} {{- end -}} {{- end -}}
